De elektrische lineaire actuator (750N / 75kg, 50mm slag) is de absolute beste keuze. Geen lekkagerisico, geen compressor, direct aanstuurbaar via Arduino L298N driver, ingebouwde eindschakelaar stopt automatisch, ingebouwde rem houdt positie vast bij stroomverlies. Beschikbaar voor ~€30–40 op AliExpress.

Hoe werkt het precies? De lineaire actuator zit parallel aan je elleboogscharnier. Als jij je bicep aanspant → EMG sensor detecteert de elektrische activiteit → Arduino leest analoog signaal → overschrijdt drempelwaarde → L298N H-bridge schakelt 12V door naar de actuator → stang strekt uit → elleboog buigt → jij tilt met 75kg ondersteuning. Bij ontspanning: actuator trekt terug op langzame PWM snelheid. Rem houdt positie automatisch vast zodra stroom wegvalt.
Strategie: Print ALTIJD eerst in goedkope PLA op 20% infill. Test de pasvorm. Pas als alles 100% klopt → definitief in PETG.

13
NEXO-1 Code Uploaden + EMG Kalibreren
Kopieer de code van de Code-pagina. Plak in Arduino IDE. Upload. Open Serial Monitor op 9600 baud. Ontspan bicep: noteer de rustwaarde. Span bicep stevig aan: noteer de activatiewaarde. Bereken en stel de drempelwaarden in.
DREMPEL_AAN = (rustwaarde + activatiewaarde) ÷ 2
DREMPEL_UIT = rustwaarde + 60

💻 nexo1.ino — Volledige Code
// ╔═══════════════════════════════════════════╗
// ║  NEXO-1 // Elektrisch Arm Exoskelet      ║
// ║  Arduino Nano + EMG Sensor + L298N       ║
// ║  Lineaire actuator 750N / 75 kg          ║
// ║  v2.1 — met anti-oscillatie + veiligheid ║
// ╚═══════════════════════════════════════════╝

// ── PIN DEFINITIE ──────────────────────────────
#define EMG_PIN       A0   // EMG sensor uitgang (0–1023)
#define ACT_IN1       7    // L298N IN1 → actuator OMHOOG (uitstrekken)
#define ACT_IN2       8    // L298N IN2 → actuator OMLAAG (intrekken)
#define ACT_ENA       9    // L298N ENA (PWM snelheidscontrole 0–255)
#define NOODSTOP_PIN  2    // NC noodstopknop (LOW = STOP ALLES)
#define LED_PIN       13   // Ingebouwde LED (HIGH = actuator actief)

// ── KALIBRATIE (pas aan na Serial Monitor meting) ──
int DREMPEL_AAN    = 420;  // EMG-waarde: spier gespannen → omhoog
int DREMPEL_UIT    = 180;  // EMG-waarde: spier ontspannen → omlaag
int SNELHEID       = 210;  // PWM 0–255 (lager = meer kracht, minder snelheid)
int SNELHEID_DOWN  = 140;  // PWM omlaag (langzamer voor veiligheid)
int VERTR_OMLAAG   = 380;  // ms: hoe lang omlaag bewegen bij ontspanning
int LOOP_DELAY     = 30;   // ms: reactiecyclus
int EMG_SAMPLES    = 8;    // Gemiddelde over N samples (anti-ruis)
int OVERHEAT_LIMIT = 180;  // seconden continue actief voor pauze

// ── STATE VARIABELEN ───────────────────────────
bool noodstopActief = false;
int  actuatorStatus  = 0;    // 0=stil, 1=omhoog, -1=omlaag
unsigned long actStartTime = 0;
unsigned long actTotalTime = 0;

// ── SETUP ──────────────────────────────────────
void setup() {
  Serial.begin(9600);
  pinMode(ACT_IN1,    OUTPUT);
  pinMode(ACT_IN2,    OUTPUT);
  pinMode(ACT_ENA,    OUTPUT);
  pinMode(LED_PIN,    OUTPUT);
  pinMode(NOODSTOP_PIN, INPUT_PULLUP);
  allesUit();

  // Startup blink sequence
  for(int i=0; i<3; i++){
    digitalWrite(LED_PIN, HIGH); delay(120);
    digitalWrite(LED_PIN, LOW);  delay(120);
  }
  Serial.println("╔══════════════════════════════╗");
  Serial.println("║  NEXO-1 v2.1 -- ONLINE      ║");
  Serial.println("║  Open Serial Monitor 9600   ║");
  Serial.println("╚══════════════════════════════╝");
  Serial.println("Kalibratie: Ontspan bicep → noteer waarde");
  Serial.println("           Span bicep aan  → noteer waarde");
}

// ── EMG UITLEZEN MET MIDDELING (anti-ruis) ────
int leesEMG() {
  long som = 0;
  for(int i=0; i<EMG_SAMPLES; i++){
    som += analogRead(EMG_PIN);
    delay(2);
  }
  return (int)(som / EMG_SAMPLES);
}

// ── HOOFD LOOP ─────────────────────────────────
void loop() {

  // 1. NOODSTOP CHECK (altijd als allereerste!)
  if (digitalRead(NOODSTOP_PIN) == LOW) {
    allesUit();
    noodstopActief = true;
    // SOS blink
    for(int i=0; i<6; i++){
      digitalWrite(LED_PIN, HIGH); delay(80);
      digitalWrite(LED_PIN, LOW);  delay(80);
    }
    Serial.println("!! NOODSTOP GEACTIVEERD -- Reset Arduino om te hervatten !!");
    while(true) { delay(50); } // volledig halt
  }

  // 2. OVERVERHITTING BEVEILIGING
  if (actuatorStatus != 0) {
    if (millis() - actStartTime > (unsigned long)OVERHEAT_LIMIT * 1000) {
      allesUit();
      Serial.println("!! OVERVERHITTING PAUZE -- 30 seconden afkoelen !!");
      delay(30000);
      actStartTime = millis();
    }
  }

  // 3. EMG SENSOR UITLEZEN (gemiddelde 8 samples)
  int emg = leesEMG();
  Serial.print("EMG:"); Serial.print(emg);
  Serial.print("  AAN>"); Serial.print(DREMPEL_AAN);
  Serial.print("  UIT<"); Serial.println(DREMPEL_UIT);

  // 4. ACTUATOR AANSTURING LOGICA
  if (emg > DREMPEL_AAN) {
    // ▲ Spier GESPANNEN → arm OMHOOG (actuator uitstrekken)
    if (actuatorStatus != 1) { actStartTime = millis(); }
    actuatorStatus = 1;
    analogWrite (ACT_ENA,  SNELHEID);
    digitalWrite(ACT_IN1, HIGH);
    digitalWrite(ACT_IN2, LOW);
    digitalWrite(LED_PIN, HIGH);

  } else if (emg < DREMPEL_UIT) {
    // ▼ Spier ONTSPANNEN → arm gecontroleerd omlaag
    actuatorStatus = -1;
    analogWrite (ACT_ENA,  SNELHEID_DOWN);
    digitalWrite(ACT_IN1, LOW);
    digitalWrite(ACT_IN2, HIGH);
    delay(VERTR_OMLAAG);
    allesUit(); // rem actief: houdt positie vast
    actuatorStatus = 0;
    digitalWrite(LED_PIN, LOW);

  } else {
    // = Neutrale zone → HOUD POSITIE (actuator ingebouwde rem)
    allesUit();
    actuatorStatus = 0;
  }

  delay(LOOP_DELAY);
}

// ── HELPER: alle uitgang uitschakelen ─────────
void allesUit() {
  digitalWrite(ACT_IN1, LOW);
  digitalWrite(ACT_IN2, LOW);
  analogWrite (ACT_ENA,  0);
  digitalWrite(LED_PIN,  LOW);
}
🎯 Kalibratie Procedure
1
Upload code → open Serial Monitor → stel in op 9600 baud
2
Ontspan bicep volledig 10 seconden → lees EMG-waarden → dit is je rustwaarde (typisch 50–200)
3
Span bicep stevig aan 5 seconden → lees piekwaarden → dit is je activatiewaarde (typisch 400–900)
4
Bereken: DREMPEL_AAN = (rust + actief) ÷ 2  ·  DREMPEL_UIT = rust + 60
5
Pas de waarden aan boven in de code → hercompileer en upload → test opnieuw
6
Pas SNELHEID aan: begin bij 180, verhoog naar max 220. Actuator mag niet oververhitten bij continu gebruik > 3 min
Pro tip: Markeer de exacte elektrode-posities op je huid met een huidvriendelijke stift. Elektroden worden minder goed na 2–3 sessies — vervang ze regelmatig voor betrouwbaar signaal. Ontvet de huid met alcohol vóór plakken voor betere geleiding.
